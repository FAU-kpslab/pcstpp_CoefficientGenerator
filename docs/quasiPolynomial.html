 75    pretty_factor_print(coeff)

352        """
353        p1 * p2 | p * scalar
354
355        Multiplies two polynomials or a polynomial with a scalar.
356
357        Returns
358        -------
359        Polynomial
360        """
361
362        # Check whether the second object is a polynomial.
363        if isinstance(other, Polynomial):
364            # Calculate the matrix containing all combinations of __private_coefficients of both polynomials.
365            # Flip it such that all __private_coefficients corresponding to the same x^n are part of the same diagonals.
366            combinations = np.flipud(np.outer(self.__private_coefficients, other.__private_coefficients))
367            # Sum over the diagonals to obtain the real __private_coefficients.
368            output: List[Coeff] = [sum(combinations.diagonal(exponent), Fraction(0)) for exponent in
369                                   np.arange(- self.__private_coefficients.size + 1, other.__private_coefficients.size)]
370            return Polynomial(output).simplify()
371        # Check whether the second object is a scalar and call scalar_multiplication.
372        elif isinstance(other, (Fraction, int, float, complex, Expr)):
373            return self.scalar_multiplication(other)
374        # If the second polynomial is not a polynomial (but e.g. a quasi-polynomial) return NotImplemented to trigger
375        # the function __rmul__ of the other class.
376        else:
377            return NotImplemented

517        """
518        qp.simplify()
519
520        Simplifies a quasi-polynomial by *removing* zero polynomials.
521
522        Returns
523        -------
524        QuasiPolynomial
525        """
526
527        output = {}
528        key_list = list(self.polynomial_dict.keys())
529        for i in range(len(key_list)):
530            # Insert the first element.
531            if len(output) == 0:
532                output[key_list[i]] = self.polynomial_dict[key_list[i]]
533            else:
534                # Check whether two exponentials e1 and e2 are almost equal.
535                if are_close(key_list[i], key_list[i - 1]):
536                    # Remove e1.
537                    output.popitem()
538                    # Add e1 + e2.
539                    output[Fraction(1, 2) * (key_list[i - 1] + key_list[i])] = self.polynomial_dict[key_list[i - 1]] + \
540                                                                               self.polynomial_dict[key_list[i]]
541                else:
542                    output[key_list[i]] = self.polynomial_dict[key_list[i]]
543        return QuasiPolynomial({e: p.simplify() for e, p in output.items() if p != Polynomial.zero()})

758
759        Returns

762        """
763
764        # Check whether the quasi-polynomial is empty.
765        if self == QuasiPolynomial.zero():
766            return QuasiPolynomial.zero()
767        else:
768            # Initiate the constant of integration.
769            constant = Fraction(0)

773                if is_zero(e):

775                else:
776                    # Give the polynomial a name to be able to differentiate it multiple times.
777                    temp_polynomial = p

784                        resulting_polynomial = resulting_polynomial - (temp_polynomial * inverse(e) ** (n + 1))
785                        resulting_constant = resulting_constant + temp_polynomial.coefficients()[0] * inverse(e) ** (
786                                n + 1)

788                    output[e] = resulting_polynomial
